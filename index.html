// ============================================================================= // Kenya HIV Facility Reporting — Google Sheets Backend v2 // Paste in: Google Sheet → Extensions → Apps Script // Deploy as Web App: Execute as Me, Access: Anyone // ============================================================================= function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Submissions') || ss.insertSheet('Submissions'); var data = JSON.parse(e.postData.contents); // Add headers if sheet is empty if (sheet.getLastRow() === 0) { var hdrs = ['Timestamp','County','Sub-County','Facility Name','MFL Code', 'Data Collector','Designation','Period','Month','Year', 'Program','Indicator','Code','Value']; sheet.appendRow(hdrs); sheet.getRange(1,1,1,hdrs.length) .setBackground('#4A235A').setFontColor('#FFFFFF') .setFontWeight('bold').setFontFamily('Arial'); sheet.setFrozenRows(1); // Set column widths [140,100,110,200,80,140,120,70,60,60,80,250,110,80] .forEach(function(w,i){ sheet.setColumnWidth(i+1,w); }); } var rows = data.rows; rows.forEach(function(row) { sheet.appendRow([ new Date(), row.county || '', row.subcounty || '', row.facility || '', row.mfl || '', row.collector || '', row.designation || '', row.period || '', row.month || '', row.year || '', row.program || '', row.indicator || '', row.code || '', row.value || '' ]); }); // Alternate row shading var last = sheet.getLastRow(); var start = last - rows.length + 1; for (var r = start; r <= last; r++) { var bg = r % 2 === 0 ? '#F5EEF8' : '#FFFFFF'; sheet.getRange(r, 1, 1, 14).setBackground(bg); } return ContentService .createTextOutput(JSON.stringify({ status: 'success', rows_saved: rows.length, facility: rows[0] ? rows[0].facility : '' })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:'error',message:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } // ============================================================================= // doGet — called by form to check if facility+period data still exists // URL: WEBAPP_URL?action=check&facility=FACILITY_NAME // Returns JSON: { submitted: ["Jan-24","Feb-24",...] } // ============================================================================= function doGet(e) { try { var action = e.parameter.action || ''; var facility = e.parameter.facility || ''; if (action === 'check' && facility) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Submissions'); if (!sheet || sheet.getLastRow() <= 1) { return ContentService .createTextOutput(JSON.stringify({submitted:[]})) .setMimeType(ContentService.MimeType.JSON); } var data = sheet.getDataRange().getValues(); var submitted = []; // Col 3 = Facility Name (0-indexed), Col 7 = Period for (var r = 1; r < data.length; r++) { if (data[r][3] && data[r][3].toString().trim() === facility.trim()) { var period = data[r][7] ? data[r][7].toString().trim() : ''; if (period && submitted.indexOf(period) === -1) { submitted.push(period); } } } return ContentService .createTextOutput(JSON.stringify({submitted: submitted, facility: facility})) .setMimeType(ContentService.MimeType.JSON); } return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:'error',message:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } // ============================================================================= // testSetup — run once to verify sheet is ready // ============================================================================= function testSetup() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Submissions') || ss.insertSheet('Submissions'); Logger.log('Sheet: ' + sheet.getName()); Logger.log('URL: ' + ss.getUrl()); }
